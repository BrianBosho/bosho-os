% ---- PUBLICATIONS ----
\section{Publication}
\begin{itemize}
    \item[] \textbf{FedProp: Federated Graph Neural Networks with Feature Propagation}
    \item[] \emph{B. Kipkirui} — Under review, major AI conference
    \item[] Cross-client edges degrade federated GNN accuracy. FedProp propagates features iteratively to reconstruct missing l-hop neighborhood information — no inter-client communication, no extra parameters.
    \item[] Result: 81.0\% accuracy on Cora (non-IID) — 19.1\% relative improvement over baseline (68.0\%).
    \item[] Includes spectral graph theory convergence guarantee.
\end{itemize}
